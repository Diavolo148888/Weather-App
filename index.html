const apiKey = "58ba2d5e4a54464f92f51820260203"; const city = "Baler"; if ("Notification" in window) { Notification.requestPermission(); } function triggerNotification(title, message) { if (Notification.permission === "granted") { new Notification(title, { body: message, tag: 'baler-update' }); } } async function getBalerWeather() { document.getElementById('dateTime').innerText = "Last update: " + new Date().toLocaleTimeString(); try { const response = await fetch(`https://api.weatherapi.com/v1/forecast.json?key=${apiKey}&q=${city}&days=1&aqi=no&alerts=no`); if (!response.ok) throw new Error("API Key Invalid"); const data = await response.json(); const tempC = Math.round(data.current.temp_c); const clouds = data.current.cloud; const humidity = data.current.humidity; const windKph = data.current.wind_kph; const condition = data.current.condition.text; const moonPhase = data.forecast.forecastday[0].astro.moon_phase; // --- UPDATED LOGIC FOR SAFETY TIPS & STATUS --- let safetyTip = ""; const lowerCondition = condition.toLowerCase(); if (lowerCondition.includes("rain") || lowerCondition.includes("shower") || lowerCondition.includes("drizzle")) { // Priority 1: Rain/Showers safetyTip = "Patchy rain nearby. 🌧️ Check the sky before heading to Sabang Beach!"; } else if (tempC >= 28 && clouds < 40) { // Priority 2: Sunny Day (Threshold lowered to 28°C) safetyTip = "Sunny and warm! ☀️ Perfect for surfing. Use SPF 30+ and stay hydrated."; } else if (windKph > 20) { // Priority 3: High Wind safetyTip = "Breezy conditions. 🚩 Watch out for choppy waves and strong currents."; } else { // Default: Steady Weather safetyTip = "Steady tropical weather. Enjoy your day in Baler! 🌴"; } // Update the User Interface document.getElementById('temp').innerText = tempC + "°C"; document.getElementById('cloudPrediction').innerText = condition; document.getElementById('clouds').innerText = clouds + "%"; document.getElementById('moon').innerText = moonPhase; document.getElementById('humidity').innerText = humidity + "%"; document.getElementById('wind').innerText = windKph + " km/h"; document.getElementById('tip').innerText = "STATUS: " + safetyTip; // Background Color Logic if (lowerCondition.includes("rain")) { document.body.style.background = "#3e4a59"; // Rainy Grey-Blue } else if (tempC >= 28 && clouds < 40) { document.body.style.background = "#1e3c72"; // Sunny Deep Blue } else { document.body.style.background = "#1a1a2e"; // Default Night/Cloudy } triggerNotification(`Baler Update: ${tempC}°C`, safetyTip); } catch (err) { console.error(err); document.getElementById('cloudPrediction').innerText = "API Error"; document.getElementById('tip').innerText = "Error: Check your API Key."; } } getBalerWeather(); // 1-Hour Auto-Sync (3,600,000 ms) setInterval(getBalerWeather, 3600000);
